Trading Bot Configuration

Here is the configuration for the trading bot:

            <TradingBotConfig>
                <!-- Initial Configuration -->
                <InitialStake value="1.0"/> <!-- Starting stake amount -->
                <StakeMultiplier value="1.0"/> <!-- Multiplier for stake adjustments -->

                <!-- Risk Management -->
                <RiskManagement>
                    <StopLossPercentage value="0.02"/> <!-- Stop loss at 2% -->
                    <RecoveryThreshold value="0.01"/> <!-- Recovery starts at 1% loss -->
                    <MaxLoss value="0.05"/> <!-- Maximum loss allowed -->
                </RiskManagement>

                <!-- Profit Management -->
                <ProfitManagement>
                    <ProfitTarget value="0.10"/> <!-- Target profit of 10% -->
                    <TrailingStop value="0.03"/> <!-- Trailing stop at 3% -->
                </ProfitManagement>

                <!-- Trade Parameters -->
                <TradeParameters>
                    <WinStreakToRecover value="3"/> <!-- Win streak to end recovery -->
                    <RecoveryStakeMultiplier value="2.0"/> <!-- Multiplier during recovery -->
                </TradeParameters>

                <!-- Trade Conditions -->
                <TradeConditions>
                    <CalculateProfitFormula>
                        <![CDATA[
                        ROUND(totalProfit * 100, 2) --> Rounds profit to 2 decimal places
                        ]]>
                    </CalculateProfitFormula>
                    <CalculateLossFormula>
                        <![CDATA[
                        ABS(totalLoss) --> Absolute value of loss
                        ]]>
                    </CalculateLossFormula>
                </TradeConditions>

                <!-- Recovery Logic -->
                <RecoveryLogic>
                    <StartRecoveryIfLossExceeds value="0.01"/> <!-- Start recovery if loss exceeds 1% -->
                    <EndRecoveryIfWinStreakExceeds value="3"/> <!-- End recovery if win streak is greater than 3 -->
                </RecoveryLogic>

                <!-- Alerts and Logging -->
                <Alerts>
                    <ProfitAlert threshold="0.10"/> <!-- Alert at 10% profit -->
                    <LossAlert threshold="0.05"/> <!-- Alert at 5% loss -->
                </Alerts>

                <!-- Additional Settings -->
                <AdditionalSettings>
                    <MaxConcurrentTrades value="1"/> <!-- Limit to 1 concurrent trade -->
                    <TradeTimeout value="60"/> <!-- Timeout for trades in seconds -->
                </AdditionalSettings>
            </TradingBotConfig>
